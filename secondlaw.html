        function isJustForce(input) {

            const answer = input.toLowerCase().trim();

            if (answer === '') return false;

            const wrongAnswers = ['force', 'the force', 'f', 'applied force', 'pushing force', 'pull force'];

            return wrongAnswers.includes(answer) || (answer.includes('force') && !answer.includes('resultant') && !answer.includes('net') && !answer.includes('total') && !answer.includes('sum') && !answer.includes('unbalanced'));

        }

        function generateQuestions() {

            questions = [];


            // Q1-3: Horizontal with specific hint requests

            [1, 2, 3].forEach(id => {

                let fList = [];

                if (id === 1) {

                    fList = [{ m: randInt(15, 45), d: 'right' }, { m: randInt(15, 45), d: 'left' }];

                } else if (id === 2) {

                    fList = [{ m: randInt(10, 25), d: 'right' }, { m: randInt(10, 25), d: 'right' }, { m: randInt(10, 25), d: 'left' }, { m: randInt(10, 25), d: 'left' }];

        function drawDiagram(question, svgId) {

            const svg = document.getElementById(svgId);

            svg.innerHTML = '';

            const width = 380, height = 150;

            svg.setAttribute('viewBox', `0 0 ${width} ${height}`);

            const centerX = width / 2, centerY = height / 2 + 10;

            const blockW = 60, blockH = 40, scale = 1.8;


            // Block

            const rect = document.createElementNS('http://www.w3.org/2000/svg', 'rect');

            rect.setAttribute('x', centerX - blockW/2); rect.setAttribute('y', centerY - blockH/2);

            rect.setAttribute('width', blockW); rect.setAttribute('height', blockH);

            rect.setAttribute('fill', '#5d2137'); rect.setAttribute('rx', '6');

            svg.appendChild(rect);


            const mass = document.createElementNS('http://www.w3.org/2000/svg', 'text');

            mass.setAttribute('x', centerX); mass.setAttribute('y', centerY + 5);

            mass.setAttribute('text-anchor', 'middle'); mass.setAttribute('fill', '#fee9e2');

            mass.setAttribute('font-family', 'Space Grotesk'); mass.setAttribute('font-size', '13');

            mass.setAttribute('font-weight', 'bold'); mass.textContent = question.mass + ' kg';

            svg.appendChild(mass);


            let rY = centerY - 12, lY = centerY - 12;


            question.forces.forEach(f => {

                const len = Math.max(f.magnitude * scale, 30);

                let sx, sy, ex, ey, lx, ly, anchor = 'middle';

                const headSize = 9;


                if (f.direction === 'right') {

                    sx = centerX + blockW/2; sy = rY; ex = sx + len; ey = sy;

                    lx = (sx + ex) / 2; 

                    // Special case for angled horizontal force label to be below

                    if (question.type === 'angled') {

                        ly = sy + 18;

                    } else {

                        ly = sy - 8;

                    }

                    rY += 18;

                } else if (f.direction === 'left') {

                    sx = centerX - blockW/2; sy = lY; ex = sx - len; ey = sy;

                    lx = (sx + ex) / 2; ly = sy - 8; lY += 18;

                } else if (f.direction === 'up') {

                    sx = centerX + 15; sy = centerY - blockH/2; ex = sx; ey = sy - len;

                    lx = sx + 18; ly = (sy + ey) / 2 + 4; anchor = 'start';

                } else if (f.direction === 'angle') {

                    const rad = f.angle * Math.PI / 180;

                    sx = centerX + blockW/2; sy = centerY - blockH/2 + 5;

                    ex = sx + len * Math.cos(rad); ey = sy - len * Math.sin(rad);

                    lx = ex + 5; ly = ey - 5; anchor = 'start';

                    

                    // Arc and Angle Labels Fixed and Centered

                    const arcR = 30;

                    const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');

                    path.setAttribute('d', `M ${sx+arcR} ${sy} A ${arcR} ${arcR} 0 0 0 ${sx+arcR*Math.cos(rad)} ${sy-arcR*Math.sin(rad)}`);

                    path.setAttribute('stroke', '#5d2137'); path.setAttribute('fill', 'none'); svg.appendChild(path);

                    

                    const aLab = document.createElementNS('http://www.w3.org/2000/svg', 'text');

                    // Position angle label centered between lines

                    const midRad = rad / 2;

                    aLab.setAttribute('x', sx + (arcR + 15) * Math.cos(midRad)); 

                    aLab.setAttribute('y', sy - (arcR + 15) * Math.sin(midRad) + 3);

                    aLab.setAttribute('font-size', '11'); aLab.setAttribute('fill', '#5d2137'); 

                    aLab.setAttribute('font-weight', 'bold');

                    aLab.textContent = f.angle + '°';

                    svg.appendChild(aLab);


                    const dotted = document.createElementNS('http://www.w3.org/2000/svg', 'line');

                    dotted.setAttribute('x1', sx); dotted.setAttribute('y1', sy); dotted.setAttribute('x2', sx + 60); dotted.setAttribute('y2', sy);

                    dotted.setAttribute('stroke', '#5d2137'); dotted.setAttribute('stroke-dasharray', '3,2'); svg.appendChild(dotted);

                }


                // Line

                const line = document.createElementNS('http://www.w3.org/2000/svg', 'line');

                line.setAttribute('x1', sx); line.setAttribute('y1', sy); line.setAttribute('x2', ex); line.setAttribute('y2', ey);

                line.setAttribute('stroke', '#ba684e'); line.setAttribute('stroke-width', '2.5'); svg.appendChild(line);


                // Arrowhead

                const angle = Math.atan2(ey - sy, ex - sx);

                const x1 = ex - headSize * Math.cos(angle - Math.PI/7);

                const y1 = ey - headSize * Math.sin(angle - Math.PI/7);

                const x2 = ex - headSize * Math.cos(angle + Math.PI/7);

                const y2 = ey - headSize * Math.sin(angle + Math.PI/7);

                const points = `${ex},${ey} ${x1},${y1} ${x2},${y2}`;


                const head = document.createElementNS('http://www.w3.org/2000/svg', 'polygon');

                head.setAttribute('points', points); head.setAttribute('fill', '#ba684e'); svg.appendChild(head);


                const lab = document.createElementNS('http://www.w3.org/2000/svg', 'text');

                lab.setAttribute('x', lx); lab.setAttribute('y', ly); lab.setAttribute('text-anchor', anchor);

                lab.setAttribute('font-family', 'Space Grotesk'); lab.setAttribute('font-weight', 'bold');

                lab.setAttribute('font-size', '11'); lab.textContent = f.magnitude + ' N';

                svg.appendChild(lab);

            });

        }

        function showHint(idx) {

            const q = questions[idx], box = document.getElementById(`hint-${q.id}`);

            box.innerHTML = `<strong>💡 Hint:</strong> ${q.hints.general}`;

            box.classList.toggle('show');

        }


        function checkQuestion(idx) {

            const q = questions[idx], state = questionStates[idx];

            if (state.completed) return;

            state.checksRemaining--;

            const resIn = document.getElementById(`resultant-${q.id}`), accIn = document.getElementById(`accel-${q.id}`);

            const rfF = document.getElementById(`rf-${q.id}`), afF = document.getElementById(`af-${q.id}`);

            const valR = parseFloat(resIn.value), valA = parseFloat(accIn.value);

            

            const rCorrect = !isNaN(valR) && Math.abs(valR - q.resultantForce) < (q.askDirection ? 0.5 : 1.5);

            const aCorrect = !isNaN(valA) && Math.abs(valA - q.acceleration) < 0.2;


            rfF.textContent = rCorrect ? '✓' : '✗'; rfF.style.color = rCorrect ? '#2ecc71' : '#e74c3c';

            afF.textContent = aCorrect ? '✓' : '✗'; afF.style.color = aCorrect ? '#2ecc71' : '#e74c3c';

            resIn.className = `question-input ${rCorrect ? 'correct' : 'incorrect'}`;

            accIn.className = `question-input ${aCorrect ? 'correct' : 'incorrect'}`;


            if (rCorrect && aCorrect) {

                state.completed = true;

                document.getElementById(`card-${q.id}`).classList.add('completed');

                document.getElementById(`checkbtn-${q.id}`).disabled = true;

                checkAllComplete();

            }

            const display = document.getElementById(`checks-${q.id}`);

            display.textContent = state.checksRemaining <= 5 ? `${state.checksRemaining} checks left` : '';

        }


        function checkAllComplete() {

            if (questionStates.every(s => s.completed)) {

                document.getElementById('completionBanner').classList.add('show');

                setTimeout(() => showCorrectAnimation(true), 800);

            }

        }
